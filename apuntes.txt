git checkout master algo.txt // Regresa a lo que habia antes del checkout

 Crear RAMAS***

git checkout cabecera  // Entro a esa nueva RAMA
git checkout master    // Vuelvo al master

git merge cabecera

 Para guardar sin dar git add . y  git commit -m "bla bla bla"  ***

 Subir proyectos a GitHub ***
git remote add origin https://github.com/mau-tico/hyperblog.git
git remote /// Va a salir que dice ORIGIN
git push origin master  // Envia a GitHub
git pull origin master  // Trae lo que haya en GitHub

 Crear LLaves ***
ssh-keygen -t rsa -b 4096 -C "tu@email.com"
eval $(ssh-agent -s)  // Para ver si el agente esta corriendo
ssh-add /root/.ssh/id_rsa    // Para enlazarlo a nuestro sistema
git remote set-url origin git@github.com:mau-tico/hyperblog.git  // Para usarlo con ssh

 Ver Logs / Tags***
git log --all --graph --decorate --oneline
alias logGit="git log --all --graph --decorate --oneline"

git push origin --tags
